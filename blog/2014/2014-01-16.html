using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;
using System.Speech.Recognition;
using System.Speech.AudioFormat;
using System.Globalization;

namespace SoundToText
{
    public class SpeechResult : Object
    {
        public double score ;
        public string text ;
        public override string ToString()
        {
 	         return string.Format("{0:00} {1}", score, text) ;
        }
    }

    public class SpeechResultArray : Object
    {
        public TimeSpan begin;
        public TimeSpan duration ;
        public SpeechResult [] options ;
        public override string ToString()
        {
            return string.Format("{0:00}:{1:00} for {2:00}s -- {3}", begin.Minutes, begin.Seconds, duration.TotalSeconds, options[0].text);
        }
    }

    public class SoundToTextASync
    {
        public static SoundToTextASync Create(string culture, string file, bool print = false)
        {
            return new SoundToTextASync(new CultureInfo(culture), file, print);
        }

        CultureInfo _info ;
        string _file;
        List<SpeechResultArray> _results;

        bool _print;
        bool _completed;

        public SpeechResultArray[] Recognized { get { return _results.OrderBy(c => c.begin).ToArray(); } }

        public SoundToTextASync(CultureInfo info, string file, bool print = false)
        {
            _info = info ;
            _file = file;
            _print = print;
            _results = new List<SpeechResultArray>();
        }

        public void RunRecognition()
        {
            using (SpeechRecognitionEngine recognizer = new SpeechRecognitionEngine(_info))
            {
                Grammar dictation = new DictationGrammar();
                dictation.Name = "Dictation Grammar";
                recognizer.LoadGrammar(dictation);
                recognizer.SetInputToWaveFile(_file);
                recognizer.SpeechRecognized += new EventHandler<SpeechRecognizedEventArgs>(recognizer_SpeechRecognized);
                recognizer.RecognizeCompleted += new EventHandler<RecognizeCompletedEventArgs>(recognizer_RecognizeCompleted);
                _completed = false;
                recognizer.RecognizeAsync(RecognizeMode.Multiple);
                while (!_completed)
                {
                    System.Threading.Thread.Sleep(5000);
                    //if (_print) Console.WriteLine("processing");
                }
            }
        }

        void recognizer_SpeechRecognized(object sender, SpeechRecognizedEventArgs e)
        {
            if (e.Result != null && e.Result.Text != null)
            {
                List<SpeechResult> res = new List<SpeechResult>();
                res.Add(new SpeechResult() { score = e.Result.Confidence, text = e.Result.Text });
                foreach (var alt in e.Result.Alternates)
                {
                    res.Add(new SpeechResult() { score = alt.Confidence, text = alt.Text });
                }

                SpeechResultArray final = new SpeechResultArray() { 
                            begin = e.Result.Audio.AudioPosition,
                            duration = e.Result.Audio.Duration,
                            options = res.ToArray() } ;
                _results.Add(final);
                if (_print) Console.WriteLine(final.ToString());
            }
        }

        // Handle the RecognizeCompleted event.
        void recognizer_RecognizeCompleted(object sender, RecognizeCompletedEventArgs e)
        {
            if (e.Error != null)
            {
                if (_print) 
                    Console.WriteLine("  Error encountered, {0}: {1}",
                            e.Error.GetType().Name, e.Error.Message);
            }
            if (e.Cancelled)
            {
                if (_print)
                    Console.WriteLine("  Operation cancelled.");
            }
            if (e.InputStreamEnded)
            {
                if (_print)
                    Console.WriteLine("  End of stream encountered.");
                _completed = true;
            }
        }
    }
}
